public class RutaDTO {

    public String origen;
    public String destino;

        return id;
    }

    public void setId(int id) {

        this.origen = origen;
    }

    public String getDestino() {
        return destino;
    }

    public void setDestino(String destino) {
        this.destino = destino;
    }
}
